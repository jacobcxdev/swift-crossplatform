```swift
struct BindingFeature {

    struct State: Equatable {
        var text: String = ""

    }
    enum Action: BindableAction {

    }

        BindingReducer()
    }

@Reducer
struct SendingFeature {
    @ObservableState
    struct State: Equatable {
        var count: Int = 0
        var effectRan: Bool = false
    }
    enum Action {
        case setCount(Int)
        case effectCompleted
    }
    var body: some ReducerOf<Self> {
        Reduce { state, action in
            switch action {
            case .setCount(let v):
                state.count = v
                return .run { send in await send(.effectCompleted) }
            case .effectCompleted:
                state.effectRan = true
                return .none
            }
        }
    }
}
```
