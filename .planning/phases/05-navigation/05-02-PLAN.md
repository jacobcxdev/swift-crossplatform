**Problem:** Skip's `NavigationStack` is non-generic. The modern TCA extension `extension NavigationStack where Data == StackState<State>.PathView` cannot compile on Android because Skip's `NavigationStack` has no `Data` type parameter. Additionally, the legacy `NavigationStackStore.body` passes `Binding<PathView>` which Skip force-casts to `Binding<[Any]>?` and crashes.

**Solution approach (per research recommendation Option A + modern API):**

Check the compilation result from Plan 01 Task 1. If the modern `NavigationStack.init(path:root:destination:)` extension (block 3) was re-guarded because it could not compile on Android:

**In NavigationStack+Observation.swift**, add an `#if os(Android)` section providing a free function or View wrapper that achieves the same API shape without extending the non-generic NavigationStack:

```swift
#if os(Android)
/// Android adapter: bridges TCA's StackState path to Skip's NavigationStack
/// Skip's NavigationStack is non-generic, so we provide a wrapper View
/// that converts Binding<Store<StackState, StackAction>> to Binding<[Any]>
@available(iOS 16, macOS 13, tvOS 16, watchOS 9, *)
public struct TCANavigationStack<State: Hashable, Action, Root: View, Destination: View>: View {
    let store: Binding<Store<StackState<State>, StackAction<State, Action>>>
    let root: Root
    let destination: (Store<State, Action>) -> Destination

    public init(
        path store: Binding<Store<StackState<State>, StackAction<State, Action>>>,
        @ViewBuilder root: () -> Root,
        @ViewBuilder destination: @escaping (Store<State, Action>) -> Destination
    ) {
        self.store = store
        self.root = root()
        self.destination = destination
    }

    public var body: some View {
        // Bridge StackState.PathView to [Any] for Skip's NavigationStack
        let pathBinding = Binding<[Any]>(
            get: {
                Array(self.store.wrappedValue.state.path.map { $0 as Any })
            },
            set: { newPath in
                let currentPath = Array(self.store.wrappedValue.state.path)
                if newPath.count < currentPath.count {
                    // Pop: find the ID at the new count boundary
                    let idToPop = currentPath[newPath.count].id
                    self.store.wrappedValue.send(.popFrom(id: idToPop))
                }
                // Push is handled by NavigationLink, not by path mutation
            }
        )
        NavigationStack(path: pathBinding) {
            self.root
                .navigationDestination(for: StackState<State>.Component.self) { component in
                    let childStore = self.store.wrappedValue.scope(
                        state: { $0.path[id: component.id]?.element ?? component.element },
                        action: { .element(id: component.id, action: $0) }
                    )
                    self.destination(childStore)
                }
        }
    }
}
#endif
```
