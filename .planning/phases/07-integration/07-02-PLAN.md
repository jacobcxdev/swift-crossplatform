**StressTests.swift (TEST-11 — two separate tests per D10):**

**Test 1: Store/Reducer throughput (pure state management):**
```swift
import Testing

extension Tag {
    @Tag static var stress: Self
}

@Test(.tags(.stress))
func storeReducerThroughput() async {
    // Simple counter reducer with .none effect
    let store = Store(initialState: StressCounter.State()) {
        StressCounter()
    }
    let iterations = 5_000
    let clock = ContinuousClock()

    let startMem = currentResidentMemoryBytes()
    let elapsed = clock.measure {
        for _ in 0..<iterations {
            store.send(.increment)
        }
    }
    let endMem = currentResidentMemoryBytes()

    // Must complete — proves >1000 mutations/sec
    #expect(elapsed < .seconds(5))
    // Memory must be bounded
    if let s = startMem, let e = endMem {
        #expect(e < s + 50 * 1024 * 1024) // <50MB growth
    }
}
```

**Test 2: Observation pipeline under load:**
```swift
@Test(.tags(.stress))
func observationCoalescingUnderLoad() async {
    let model = StressModel() // @Observable class with 3 properties
    var changeCount = 0
    let iterations = 5_000

    for _ in 0..<iterations {
        withObservationTracking {
            _ = model.value
        } onChange: {
            changeCount += 1
        }
        model.value += 1
    }

    // Each cycle: track → mutate → one onChange. Total = iterations.
    #expect(changeCount == iterations)
    // Memory bounded — no unbounded accumulation
}
```

Include the cross-platform `currentResidentMemoryBytes()` helper function from research (R3b §3) using `mach_task_basic_info` on Darwin and `/proc/self/status` on Linux/Android.
