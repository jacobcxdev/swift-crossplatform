<objective>
Validate CasePaths and CustomDump work on Android, adding minimal platform guards where needed and comprehensive tests.

Purpose: CasePaths provides the enum routing infrastructure TCA depends on. CustomDump provides value dumping/diffing used by TCA's test assertions. Both must work on Android. CasePaths' @CasePathable macro output is pure Swift (zero risk), but EnumMetadata ABI pointer arithmetic is used by 6 TCA core files and needs runtime verification. CustomDump's Mirror usage is low risk (stdlib APIs), but Foundation conformances and Apple-only framework conformances need canImport guards. This plan depends on 02-02 because CustomDump's expectNoDifference/expectDifference use reportIssue() which must work on Android first.

Output: Both libraries compile and pass tests on Android. EnumMetadata smoke test confirms ABI compatibility. CustomDump conformances properly guarded.
</objective>

  **Step 3: Add guards ONLY where needed**

  Per CONTEXT.md decisions:
  - Use `#if canImport(Framework)` for Apple-only framework conformances
  - Use inline guards, no separate platform files
  - Only add guards for code that would fail to compile on Android
  - Document every gated-out API for future compatibility tracking

  Example fix pattern:
  ```swift
  // Before (if missing guard):
  extension NSAttributedString: CustomDumpStringConvertible { ... }

  // After:
  #if canImport(Foundation) && !os(Android)
  extension NSAttributedString: CustomDumpStringConvertible { ... }
  #endif
  ```

  **Important:** Do NOT change guards that already work correctly. The `#if !os(WASI)` guards that include Android are intentional -- those types (Data, Date, etc.) exist in swift-corelibs-foundation and should work on Android.

      #expect(output.contains("id: 1"))
      #expect(output.contains("name: \"Blob\""))
      #expect(output.contains("email: Optional(\"blob@example.com\")") ||

  @Test func customDumpEnumOutput() {
      let status = Status.inactive(reason: "expired")
      var output = ""
      customDump(status, to: &output)
      #expect(output.contains("inactive"))
      #expect(output.contains("expired"))
  }

  @Test func customDumpCollectionOutput() {
      let numbers = [1, 2, 3]
      var output = ""
      customDump(numbers, to: &output)
      #expect(output.contains("1"))
      #expect(output.contains("2"))
      #expect(output.contains("3"))
  }

  @Test func customDumpOptionalNil() {
      let value: String? = nil
      var output = ""
      customDump(value, to: &output)
      #expect(output.contains("nil"))
  }

  // CD-02: String(customDumping:) creates string from value dump
  @Test func stringCustomDumping() {
      let user = User(id: 1, name: "Blob", email: nil)
      let output = String(customDumping: user)
      #expect(output.contains("User("))
      #expect(output.contains("id: 1"))
      #expect(output.contains("name: \"Blob\""))
  }

  // CD-03: diff computes string diff between two values
  @Test func diffDetectsChanges() {
      let user1 = User(id: 1, name: "Blob", email: nil)
      let user2 = User(id: 1, name: "Blob Jr.", email: "jr@example.com")
      let result = diff(user1, user2)
      #expect(result != nil)
      #expect(result!.contains("name"))
      #expect(result!.contains("Blob Jr."))
  }

  @Test func diffReturnsNilForEqualValues() {
      let user1 = User(id: 1, name: "Blob", email: nil)
      let user2 = User(id: 1, name: "Blob", email: nil)
      let result = diff(user1, user2)
      #expect(result == nil)
  }

  @Test func diffEnumChanges() {
      let s1 = Status.active
      let s2 = Status.inactive(reason: "expired")
      let result = diff(s1, s2)
      #expect(result != nil)
  }
